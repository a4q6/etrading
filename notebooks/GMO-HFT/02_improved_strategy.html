    # UTC
    hourly_spread = rate.groupby(rate.timestamp.dt.hour).spread_bp.mean()

    print(hourly_spread.round(2))
    
    del rate
    gc.collect()

class VolatilityAdaptiveMM_v2(StrategyBase):
    """
    GMOMarket Making
    
    :
    1. GMO
    2. 
    3. 
    """
    def __init__(
        self,
        sym: str,
        venue: str,
        amount: float,
        client: BacktestClient,
        # GMO
        base_offset: float = 5.0,  # bps- GMO
        vol_multiplier: float = 0.8,  # 
        min_offset: float = 3.0,  #  - 
        max_offset: float = 15.0,  # 
        # 
        vol_window: int = 100,  # tick
        # 
        max_position: float = 0.01,  # 
        position_decay: float = 1.5,  # 
        # 
        use_reversal: bool = True,
        reversal_lookback: int = 20,
        reversal_threshold: float = 8.0,  # 
        reversal_factor: float = 0.5,
        # 
        min_expected_profit_bp: float = 1.0,  # 
        model_id="VolAdaptiveMM_GMO_v2",
    ):
        super().__init__(model_id, log_file=None)
        self.sym = sym
        self.venue = venue
        self.client = client
        self.amount = amount
        
        self.base_offset = base_offset
        self.vol_multiplier = vol_multiplier
        self.min_offset = min_offset
        self.max_offset = max_offset
        self.vol_window = vol_window
        self.max_position = max_position
        self.position_decay = position_decay
        
        self.use_reversal = use_reversal
        self.reversal_lookback = reversal_lookback
        self.reversal_threshold = reversal_threshold
        self.reversal_factor = reversal_factor
        self.min_expected_profit_bp = min_expected_profit_bp
        
        # 
        self.mid = np.nan
        self.price_hist = deque(maxlen=300)
        self.return_hist = deque(maxlen=300)
        self.bid_order = Order.null_order()
        self.ask_order = Order.null_order()
        self.current_vol = np.nan
        self.current_spread_bp = np.nan
        
    @property
    def cur_pos(self):
        return self.client.positions.get(self.sym, [0, 0])[1]
    
    def _calc_volatility(self) -> float:
        """bps"""
        if len(self.return_hist) < 20:
            return 5.0  # 
        returns = list(self.return_hist)[-self.vol_window:]
        return np.std(returns) if len(returns) > 1 else 5.0
    
    def _get_reversal_signal(self) -> float:
        """"""
        if not self.use_reversal or len(self.price_hist) < self.reversal_lookback + 1:
            return 0.0
        prices = list(self.price_hist)
        recent_return = (prices[-1] / prices[-self.reversal_lookback] - 1) * 1e4
        if abs(recent_return) > self.reversal_threshold:
            return -recent_return * self.reversal_factor / 10
        return 0.0
    
    def _calc_dynamic_offset(self) -> float:
        """"""
        vol = self._calc_volatility()
        self.current_vol = vol
        
        #  + 
        offset = self.base_offset + vol * self.vol_multiplier
        
        # 
        if not np.isnan(self.current_spread_bp):
            # 
            spread_adjustment = max(0, self.current_spread_bp - 3.0) * 0.3
            offset += spread_adjustment
        
        return np.clip(offset, self.min_offset, self.max_offset)
    
    async def on_message(self, msg):
        dtype = msg.get("_data_type")
        
        if dtype == "Order":
            if msg["order_id"] == self.bid_order.order_id:
                self.bid_order = Order(**{k: v for k, v in msg.items() if k != "_data_type"})
            if msg["order_id"] == self.ask_order.order_id:
                self.ask_order = Order(**{k: v for k, v in msg.items() if k != "_data_type"})
                
        elif dtype == "Rate" and msg["venue"] == self.venue and msg["sym"] == self.sym:
            # 
            if msg["best_ask"] > 0 and msg["best_bid"] > 0:
                self.current_spread_bp = (msg["best_ask"] - msg["best_bid"]) / msg["mid_price"] * 1e4
            
            # 
            if not np.isnan(self.mid):
                ret = (msg["mid_price"] / self.mid - 1) * 1e4
                self.return_hist.append(ret)
            self.mid = msg["mid_price"]
            self.price_hist.append(self.mid)
            
            await self._update_quotes(msg)
    
    async def _update_quotes(self, msg):
        if np.isnan(self.mid) or len(self.price_hist) < 30:
            return
        if len(self.client.pending_limit_orders) > 0:
            return
        
        # 
        base_offset = self._calc_dynamic_offset()
        
        # 
        reversal_signal = self._get_reversal_signal()
        
        # 
        pos_ratio = self.cur_pos / self.max_position
        pos_skew = -pos_ratio * self.position_decay * 5
        
        total_skew = reversal_signal + pos_skew
        
        bid_offset = max(self.min_offset, base_offset - total_skew)
        ask_offset = max(self.min_offset, base_offset + total_skew)
        
        new_bid = np.floor(self.mid * (1 - bid_offset / 1e4))
        new_ask = np.ceil(self.mid * (1 + ask_offset / 1e4))
        
        can_buy = self.cur_pos < self.max_position
        can_sell = self.cur_pos > -self.max_position
        
        # Bid
        if can_buy:
            if not self.bid_order.is_live or abs(new_bid - self.bid_order.price) > 1:
                if self.bid_order.is_live:
                    self.bid_order = await self.client.cancel_order(
                        self.bid_order.order_id, timestamp=msg["timestamp"],
                        src_type="Rate", src_timestamp=msg["timestamp"], src_id=msg.get("universal_id", ""))
                if not self.bid_order.is_live:
                    self.bid_order = await self.client.send_order(
                        msg["timestamp"], self.sym, side=1, price=new_bid, amount=self.amount,
                        order_type=OrderType.Limit, src_type="Rate", src_timestamp=msg["timestamp"],
                        src_id=msg.get("universal_id", ""), misc="bid")
        elif self.bid_order.is_live:
            self.bid_order = await self.client.cancel_order(
                self.bid_order.order_id, timestamp=msg["timestamp"],
                src_type="Rate", src_timestamp=msg["timestamp"], src_id=msg.get("universal_id", ""))
        
        # Ask
        if can_sell:
            if not self.ask_order.is_live or abs(new_ask - self.ask_order.price) > 1:
                if self.ask_order.is_live:
                    self.ask_order = await self.client.cancel_order(
                        self.ask_order.order_id, timestamp=msg["timestamp"],
                        src_type="Rate", src_timestamp=msg["timestamp"], src_id=msg.get("universal_id", ""))
                if not self.ask_order.is_live:
                    self.ask_order = await self.client.send_order(
                        msg["timestamp"], self.sym, side=-1, price=new_ask, amount=self.amount,
                        order_type=OrderType.Limit, src_type="Rate", src_timestamp=msg["timestamp"],
                        src_id=msg.get("universal_id", ""), misc="ask")
        elif self.ask_order.is_live:
            self.ask_order = await self.client.cancel_order(
                self.ask_order.order_id, timestamp=msg["timestamp"],
                src_type="Rate", src_timestamp=msg["timestamp"], src_id=msg.get("universal_id", ""))

async def prepare_messages(st, et, venue="gmo", sym="BTCJPY"):
    """"""
    rate = load_data(date=[st, et], table="Rate", venue=venue, symbol=sym, log_output=False)
    trades = load_data(date=[st, et], table="MarketTrade", venue=venue, symbol=sym, log_output=False)
    book = load_data(date=[st, et], table="MarketBook", venue=venue, symbol=sym, log_output=False)
    
    rate["market_created_timestamp"] = rate["market_created_timestamp"].fillna(rate["timestamp"])
    rate_1s = rate.set_index("market_created_timestamp")[["best_bid", "best_ask", "sym", "venue"]].resample("1s").last().dropna().reset_index()
    
    bt_rate = list(zip(rate_1s.market_created_timestamp, rate_1s.assign(_data_type="BT_Rate").to_dict(orient="records")))
    bt_trade = list(zip(trades.market_created_timestamp, trades.assign(_data_type="BT_MarketTrade").to_dict(orient="records")))
    bt_book = list(zip(book.market_created_timestamp, book.assign(_data_type="BT_MarketBook").to_dict(orient="records")))
    rate_feed = list(zip(rate.timestamp, rate.assign(_data_type="Rate").to_dict(orient="records")))
    
    messages = sorted(bt_book + bt_rate + bt_trade + rate_feed, key=lambda x: x[0])
    return messages


async def run_backtest(messages, venue, sym, params):
    """"""
    client = BacktestClient(venue)
    strategy = VolatilityAdaptiveMM_v2(
        sym=sym, venue=venue, amount=params.get("amount", 0.001), client=client, 
        **{k: v for k, v in params.items() if k != "amount"}
    )
    client.register_strategy(strategy)
    strategy.logger.setLevel(logging.CRITICAL)
    
    # MarketBook_latest_book
    for t, msg in messages:
        if msg.get("_data_type") == "BT_MarketBook":
            client._latest_book = msg
            break
    
    for t, msg in messages:
        await client.on_message(msg)
        await strategy.on_message(msg)
    
    return client, strategy


def analyze_results(client, maker_fee_bp=0.0):
    """GMO: Maker -1bp"""
    if len(client.transactions_frame) == 0:
        return None
    
    mat = calc_matching_table(client.transactions_frame)
    if len(mat) == 0:
        return None
    
    # GMO: Maker
    mat["fee_bp"] = maker_fee_bp * 2  # Maker
    mat["pl_net"] = mat.pl_bp + mat.fee_bp
    mat["pl_jpy"] = mat.pl_net * mat.amount * mat.rinc_price / 1e4
    
    days = (mat.timestamp.max() - mat.timestamp.min()).days + 1
    
    stats = {
        "trades": len(mat),
        "daily_trades": len(mat) / days,
        "total_pl_bps": mat.pl_net.sum(),
        "avg_pl_bps": mat.pl_net.mean(),
        "win_rate": (mat.pl_net > 0).mean(),
        "max_dd_bps": mat.pl_net.cumsum().cummax().sub(mat.pl_net.cumsum()).max(),
        "avg_horizon": mat.horizon.mean(),
        "total_pl_jpy": mat.pl_jpy.sum(),
    }
    
    return mat, stats
