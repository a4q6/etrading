def extract_book_features(row, n_levels=5):
    """"""
    bids = row['bids']
    asks = row['asks']
    
    best_bid = bids[0][0] if len(bids) > 0 else np.nan
    best_ask = asks[0][0] if len(asks) > 0 else np.nan
    best_bid_size = bids[0][1] if len(bids) > 0 else 0
    best_ask_size = asks[0][1] if len(asks) > 0 else 0
    
    # n_levels
    bid_volumes = [bids[i][1] if i < len(bids) else 0 for i in range(n_levels)]
    ask_volumes = [asks[i][1] if i < len(asks) else 0 for i in range(n_levels)]
    
    total_bid_vol = sum(bid_volumes)
    total_ask_vol = sum(ask_volumes)
    
    # Order Imbalance
    imbalance = (total_bid_vol - total_ask_vol) / (total_bid_vol + total_ask_vol) if (total_bid_vol + total_ask_vol) > 0 else 0
    l1_imbalance = (best_bid_size - best_ask_size) / (best_bid_size + best_ask_size) if (best_bid_size + best_ask_size) > 0 else 0
    
    return {
        'timestamp': row['timestamp'],
        'best_bid': best_bid,
        'best_ask': best_ask,
        'best_bid_size': best_bid_size,
        'best_ask_size': best_ask_size,
        'imbalance': imbalance,
        'l1_imbalance': l1_imbalance
    }

@dataclass 
class MMStrategy:
    """FeeMM"""
    # 
    order_size: float = 100.0  # XRP
    imbalance_threshold: float = 0.3
    max_position: float = 500.0  # 
    position_timeout_sec: float = 30.0  # 
    
    # 
    position: float = 0.0
    position_entry_time: Optional[pd.Timestamp] = None
    position_entry_price: float = 0.0
    active_order: Optional[Order] = None
    
    # 
    orders: List[Order] = field(default_factory=list)
    trades: List[Trade] = field(default_factory=list)
    
    # 
    order_counter: int = 0
    trade_counter: int = 0
    
    def generate_signal(self, imbalance: float) -> int:
        """: 1=, -1=, 0="""
        if imbalance > self.imbalance_threshold:
            return 1  # 
        elif imbalance < -self.imbalance_threshold:
            return -1  # 
        return 0
    
    def should_close_position(self, now: pd.Timestamp) -> bool:
        """"""
        if self.position == 0:
            return False
        if self.position_entry_time is None:
            return False
        elapsed = (now - self.position_entry_time).total_seconds()
        return elapsed > self.position_timeout_sec
    
    def place_order(self, timestamp: pd.Timestamp, side: int, price: float) -> Order:
        """"""
        self.order_counter += 1
        order = Order(
            order_id=f"ORD_{self.order_counter:06d}",
            timestamp=timestamp,
            side=side,
            price=price,
            size=self.order_size
        )
        self.orders.append(order)
        self.active_order = order
        return order
    
    def cancel_order(self):
        """"""
        if self.active_order:
            self.active_order.status = 'canceled'
            self.active_order = None
    
    def fill_order(self, timestamp: pd.Timestamp, is_maker: bool = True):
        """"""
        if self.active_order is None:
            return
        
        order = self.active_order
        order.status = 'filled'
        order.filled_timestamp = timestamp
        
        self.trade_counter += 1
        fee_bps = MAKER_FEE_BPS if is_maker else TAKER_FEE_BPS
        
        trade = Trade(
            trade_id=f"TRD_{self.trade_counter:06d}",
            order_id=order.order_id,
            timestamp=timestamp,
            side=order.side,
            price=order.price,
            size=order.size,
            fee_bps=fee_bps
        )
        self.trades.append(trade)
        
        # 
        prev_position = self.position
        self.position += order.side * order.size
        
        # 
        if prev_position == 0 and self.position != 0:
            self.position_entry_time = timestamp
            self.position_entry_price = order.price
        elif self.position == 0:
            self.position_entry_time = None
            self.position_entry_price = 0
        
        self.active_order = None
    
    def check_fill(self, market_trade_price: float, market_trade_side: int) -> bool:
        """: """
        if self.active_order is None:
            return False
        
        order = self.active_order
        
        # : 
        if order.side == 1 and market_trade_side == -1:
            if market_trade_price <= order.price:
                return True
        
        # : 
        if order.side == -1 and market_trade_side == 1:
            if market_trade_price >= order.price:
                return True
        
        return False

def run_backtest(feat_df: pd.DataFrame, trade_df: pd.DataFrame, 
                 imbalance_threshold: float = 0.3,
                 order_size: float = 100.0,
                 position_timeout_sec: float = 30.0) -> MMStrategy:
    """"""
    
    strategy = MMStrategy(
        order_size=order_size,
        imbalance_threshold=imbalance_threshold,
        position_timeout_sec=position_timeout_sec
    )
    
    # 
    book_events = [(row['timestamp'], 'book', row) for _, row in feat_df.iterrows()]
    trade_events = [(row['timestamp'], 'trade', row) for _, row in trade_df.iterrows()]
    
    all_events = sorted(book_events + trade_events, key=lambda x: x[0])
    
    current_book = None
    
    for timestamp, event_type, data in all_events:
        
        if event_type == 'book':
            current_book = data
            
            # 
            if strategy.should_close_position(timestamp):
                # Taker
                if strategy.position > 0:
                    #   
                    strategy.cancel_order()
                    strategy.place_order(timestamp, -1, data['best_bid'])
                    strategy.fill_order(timestamp, is_maker=False)  # Taker
                elif strategy.position < 0:
                    #   
                    strategy.cancel_order()
                    strategy.place_order(timestamp, 1, data['best_ask'])
                    strategy.fill_order(timestamp, is_maker=False)  # Taker
            
            # 
            if strategy.active_order is None and abs(strategy.position) < strategy.max_position:
                signal = strategy.generate_signal(data['imbalance'])
                
                if signal == 1:  # 
                    strategy.place_order(timestamp, 1, data['best_bid'])
                elif signal == -1:  # 
                    strategy.place_order(timestamp, -1, data['best_ask'])
        
        elif event_type == 'trade':
            # 
            # sidestr
            trade_side = data['side']
            if isinstance(trade_side, str):
                trade_side = 1 if trade_side == 'buy' else -1
            
            if strategy.check_fill(data['price'], trade_side):
                strategy.fill_order(timestamp, is_maker=True)
    
    return strategy

    # P&L
    def calc_round_trip_pnl(group):
        if len(group) < 2:
            return pd.Series({'pnl_price': np.nan, 'pnl_fee': np.nan, 'pnl_total': np.nan})
        
        # 
        entry = group.iloc[0]

        if entry['side'] == 1:  #

            pnl_price = (entry['price'] - exit_['price']) * entry['size']
        
        # Fee
        pnl_fee = -group['fee_jpy'].sum()
        
        return pd.Series({
            'entry_time': entry['timestamp'],
            'exit_time': exit_['timestamp'],
            'entry_side': entry['side'],
            'entry_price': entry['price'],
            'exit_price': exit_['price'],
            'size': entry['size'],
            'pnl_price': pnl_price,
            'pnl_fee': pnl_fee,
            'pnl_total': pnl_price + pnl_fee,
