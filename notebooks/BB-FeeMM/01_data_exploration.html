sample_date = '2026-01-25'
spread_data = []

for sym in symbols:
    try:
        df = load_data(sample_date, 'Rate', venue, sym)
        if df is not None and len(df) > 0:
            # bps (: best_bid, best_ask, mid_price)
            df['spread_bps'] = (df['best_ask'] - df['best_bid']) / df['mid_price'] * 10000
            
            spread_data.append({
                'symbol': sym,
                'spread_mean_bps': df['spread_bps'].mean(),

    except Exception as e:
        print(f"{sym}: {e}")

    spread_df = spread_df.sort_values('spread_mean_bps')

# MarketBook
book_data = []

for sym in symbols:
    try:
        df = load_data(sample_date, 'MarketBook', venue, sym)
        if df is not None and len(df) > 0:
            # Best Bid/Ask  (bids/asks[price, size]numpy array)
            best_bid_sizes = []
            best_ask_sizes = []
            
            for _, row in df.iterrows():
                if 'bids' in df.columns and len(row['bids']) > 0:
                    best_bid_sizes.append(row['bids'][0][1])  # [price, size]
                if 'asks' in df.columns and len(row['asks']) > 0:
                    best_ask_sizes.append(row['asks'][0][1])
            
            if best_bid_sizes and best_ask_sizes:
                book_data.append({
                    'symbol': sym,
                    'best_bid_size_mean': np.mean(best_bid_sizes),
                    'best_ask_size_mean': np.mean(best_ask_sizes),
                    'best_bid_size_median': np.median(best_bid_sizes),
                    'best_ask_size_median': np.median(best_ask_sizes),
                    'sample_count': len(df)
                })
    except Exception as e:
        print(f"{sym}: {e}")

if book_data:
    book_df = pd.DataFrame(book_data)
    print(f"{sample_date}")
    display(book_df.round(4))
else:
    print("MarketBook")

            })
