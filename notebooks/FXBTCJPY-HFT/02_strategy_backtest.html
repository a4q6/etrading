# Cross-Market Signal MM
class CrossMarketMM(StrategyBase):
    """
    Binance/BitMex/BTCJPYMM
    
    :
    - global_mid = mean(mid_price * USDJPY)
    - signal = (global_mid / prev_global_mid - 1) 
    - signalBidAsk
    """
    def __init__(
        self,
        sym: str,
        venue: str,
        amount: float,
        client: BacktestClient,
        # 
        signal_lookback: int = 60,  # lookback
        signal_threshold: float = 1.0,  # bps
        # 
        base_offset: float = 0.5,  # bps
        skew_factor: float = 1.0,  # 
        # 
        max_position: float = 0.01,  # BTC
        position_decay: float = 0.5,  # 
        # 
        reference = [("binance", "BTCUSDT"), ("bitflyer", "BTCJPY"), ("bitmex", "XBTUSD")],
        # UTC- JST 19:00-19:20
        itayose_filter: tuple = (10, 0, 20),  # (hour, start_min, end_min)
        model_id="CrossMarketMM",
    ):
        super().__init__(model_id, log_file=None)
        self.sym = sym
        self.venue = venue
        self.client = client
        self.amount = amount
        self.signal_lookback = signal_lookback
        self.signal_threshold = signal_threshold
        self.base_offset = base_offset
        self.skew_factor = skew_factor
        self.max_position = max_position
        self.position_decay = position_decay
        self.reference = reference
        self.itayose_filter = itayose_filter
        
        # 
        self.usdjpy: float = np.nan
        self.mid: float = np.nan
        self.latest_rate: Dict = {}
        self.latest_mid: Dict[Tuple[str, str], float] = {k: np.nan for k in reference}
        self.gmid_hist: deque = deque(maxlen=1000)
        self.last_gmid: float = np.nan
        
        self.bid_order: Order = Order.null_order()
        self.ask_order: Order = Order.null_order()
        self._last_quote_time = pd.Timestamp("2000-01-01", tz="UTC")
        
    @property
    def cur_pos(self):
        return self.client.positions.get(self.sym, [0, 0])[1]
    
    def _is_itayose_time(self, ts) -> bool:
        """JST 19:00-19:20"""
        hour, start_min, end_min = self.itayose_filter
        if ts.hour == hour and start_min <= ts.minute <= end_min:
            return True
        return False
    
    def _get_signal(self, now) -> float:
        """bps"""
        if len(self.gmid_hist) < 2:
            return 0.0
        
        # lookbackgmid
        lookback_time = now - datetime.timedelta(seconds=self.signal_lookback)
        past_gmid = None
        for t, g in self.gmid_hist:
            if t <= lookback_time:
                past_gmid = g
            else:
                break
        
        if past_gmid is None or past_gmid == 0:
            return 0.0
        
        current_gmid = self.gmid_hist[-1][1]
        signal = (current_gmid / past_gmid - 1) * 1e4  # bps
        return signal
    
    async def on_message(self, msg: Dict):
        dtype = msg.get("_data_type")
        
        if dtype == "Order":
            await self._handle_order(msg)
        elif dtype == "Rate":
            self._update_prices(msg)
            await self._update_quotes(msg)
    
    async def _handle_order(self, msg: Dict):
        if msg["order_id"] == self.bid_order.order_id:
            self.bid_order = Order(**{k: v for k, v in msg.items() if k != "_data_type"})
        if msg["order_id"] == self.ask_order.order_id:
            self.ask_order = Order(**{k: v for k, v in msg.items() if k != "_data_type"})
    
    def _update_prices(self, msg: Dict):
        # 
        if self._is_itayose_time(msg["timestamp"]):
            return
            
        if msg["sym"] == "USDJPY":
            self.usdjpy = msg["mid_price"]
        elif msg["venue"] == self.venue and msg["sym"] == self.sym:
            self.mid = msg["mid_price"]
            self.latest_rate = msg
        else:
            key = (msg["venue"], msg["sym"])
            if key in self.latest_mid:
                if "USD" in msg["sym"] and not np.isnan(self.usdjpy):
                    self.latest_mid[key] = msg["mid_price"] * self.usdjpy
                else:
                    self.latest_mid[key] = msg["mid_price"]
                
                # gmid
                if all(not np.isnan(v) for v in self.latest_mid.values()):
                    gmid = np.mean(list(self.latest_mid.values()))
                    self.gmid_hist.append((msg["timestamp"], gmid))
                    self.last_gmid = gmid
    
    async def _update_quotes(self, msg: Dict):
        if msg["venue"] != self.venue or msg["sym"] != self.sym:
            return
        if np.isnan(self.mid) or len(self.gmid_hist) < 10:
            return
        # pending_limit_orders
        if len(self.client.pending_limit_orders) > 0:
            return
        
        now = msg["timestamp"]
        
        # JST 19:00-19:20 = UTC 10:00-10:20
        if self._is_itayose_time(now):
            if self.bid_order.is_live:
                await self._cancel_bid(msg)
            if self.ask_order.is_live:
                await self._cancel_ask(msg)
            return
        
        # 
        signal = self._get_signal(now)
        
        # 
        pos_skew = -self.cur_pos / self.max_position * self.position_decay * 10  # bps
        
        # 
        total_skew = signal * self.skew_factor + pos_skew
        
        bid_offset = self.base_offset - total_skew
        ask_offset = self.base_offset + total_skew
        
        new_bid = np.floor(self.mid * (1 - bid_offset / 1e4))
        new_ask = np.ceil(self.mid * (1 + ask_offset / 1e4))
        
        # 
        can_buy = self.cur_pos < self.max_position
        can_sell = self.cur_pos > -self.max_position
        
        # 
        strong_signal = abs(signal) > self.signal_threshold
        
        # Bid
        if can_buy and (signal > 0 or not strong_signal):
            await self._update_bid(msg, new_bid)
        elif self.bid_order.is_live:
            await self._cancel_bid(msg)
        
        # Ask
        if can_sell and (signal < 0 or not strong_signal):
            await self._update_ask(msg, new_ask)
        elif self.ask_order.is_live:
            await self._cancel_ask(msg)
    
    async def _update_bid(self, msg, new_bid):
        need_update = (
            not self.bid_order.is_live or
            abs(new_bid - self.bid_order.price) > 1
        )
        if need_update:
            if self.bid_order.is_live:
                self.bid_order = await self.client.cancel_order(
                    self.bid_order.order_id, timestamp=msg["timestamp"],
                    src_type="Rate", src_timestamp=msg["timestamp"], src_id=msg.get("universal_id", ""))
            if not self.bid_order.is_live:
                self.bid_order = await self.client.send_order(
                    msg["timestamp"], self.sym, side=1, price=new_bid, amount=self.amount,
                    order_type=OrderType.Limit, src_type="Rate", src_timestamp=msg["timestamp"],
                    src_id=msg.get("universal_id", ""), misc="bid")
    
    async def _update_ask(self, msg, new_ask):
        need_update = (
            not self.ask_order.is_live or
            abs(new_ask - self.ask_order.price) > 1
        )
        if need_update:
            if self.ask_order.is_live:
                self.ask_order = await self.client.cancel_order(
                    self.ask_order.order_id, timestamp=msg["timestamp"],
                    src_type="Rate", src_timestamp=msg["timestamp"], src_id=msg.get("universal_id", ""))
            if not self.ask_order.is_live:
                self.ask_order = await self.client.send_order(
                    msg["timestamp"], self.sym, side=-1, price=new_ask, amount=self.amount,
                    order_type=OrderType.Limit, src_type="Rate", src_timestamp=msg["timestamp"],
                    src_id=msg.get("universal_id", ""), misc="ask")
    
    async def _cancel_bid(self, msg):
        self.bid_order = await self.client.cancel_order(
            self.bid_order.order_id, timestamp=msg["timestamp"],
            src_type="Rate", src_timestamp=msg["timestamp"], src_id=msg.get("universal_id", ""))
    
    async def _cancel_ask(self, msg):
        self.ask_order = await self.client.cancel_order(
            self.ask_order.order_id, timestamp=msg["timestamp"],
            src_type="Rate", src_timestamp=msg["timestamp"], src_id=msg.get("universal_id", ""))

# Order Flow Imbalance MM
class OrderFlowMM(StrategyBase):
    """
    MarketTradeOrder Flow ImbalanceMM
    
    :
    - OFI = ( - ) / 
    - OFIAsk
    - OFIBid
    """
    def __init__(
        self,
        sym: str,
        venue: str,
        amount: float,
        client: BacktestClient,
        # OFI
        ofi_window: int = 60,  # OFI
        ofi_threshold: float = 0.3,  # OFI
        # 
        base_offset: float = 0.5,
        skew_factor: float = 2.0,
        # 
        max_position: float = 0.01,
        position_decay: float = 0.5,
        # UTC- JST 19:00-19:20
        itayose_filter: tuple = (10, 0, 20),  # (hour, start_min, end_min)
        model_id="OrderFlowMM",
    ):
        super().__init__(model_id, log_file=None)
        self.sym = sym
        self.venue = venue
        self.client = client
        self.amount = amount
        self.ofi_window = ofi_window
        self.ofi_threshold = ofi_threshold
        self.base_offset = base_offset
        self.skew_factor = skew_factor
        self.max_position = max_position
        self.position_decay = position_decay
        self.itayose_filter = itayose_filter
        
        # 
        self.mid: float = np.nan
        self.latest_rate: Dict = {}
        self.trade_hist: deque = deque(maxlen=10000)
        
        self.bid_order: Order = Order.null_order()
        self.ask_order: Order = Order.null_order()
        
    @property
    def cur_pos(self):
        return self.client.positions.get(self.sym, [0, 0])[1]
    
    def _is_itayose_time(self, ts) -> bool:
        """JST 19:00-19:20"""
        hour, start_min, end_min = self.itayose_filter
        if ts.hour == hour and start_min <= ts.minute <= end_min:
            return True
        return False
    
    def _get_ofi(self, now) -> float:
        """Order Flow Imbalance-1 ~ +1"""
        if len(self.trade_hist) == 0:
            return 0.0
        
        cutoff = now - datetime.timedelta(seconds=self.ofi_window)
        buy_vol = 0.0
        sell_vol = 0.0
        
        for t, side, amount in self.trade_hist:
            if t < cutoff:
                continue
            if side > 0:
                buy_vol += amount
            else:
                sell_vol += amount
        
        total = buy_vol + sell_vol
        if total == 0:
            return 0.0
        
        return (buy_vol - sell_vol) / total
    
    async def on_message(self, msg: Dict):
        dtype = msg.get("_data_type")
        
        if dtype == "Order":
            await self._handle_order(msg)
        elif dtype == "Rate":
            if msg["venue"] == self.venue and msg["sym"] == self.sym:
                # 
                if not self._is_itayose_time(msg["timestamp"]):
                    self.mid = msg["mid_price"]
                    self.latest_rate = msg
                await self._update_quotes(msg)
        elif dtype == "BT_MarketTrade":
            # 
            if not self._is_itayose_time(msg["market_created_timestamp"]):
                self.trade_hist.append((msg["market_created_timestamp"], msg["side"], msg["amount"]))
    
    async def _handle_order(self, msg: Dict):
        if msg["order_id"] == self.bid_order.order_id:
            self.bid_order = Order(**{k: v for k, v in msg.items() if k != "_data_type"})
        if msg["order_id"] == self.ask_order.order_id:
            self.ask_order = Order(**{k: v for k, v in msg.items() if k != "_data_type"})
    
    async def _update_quotes(self, msg: Dict):
        if np.isnan(self.mid):
            return
        # pending_limit_orders
        if len(self.client.pending_limit_orders) > 0:
            return
        
        now = msg["timestamp"]
        
        # JST 19:00-19:20 = UTC 10:00-10:20
        if self._is_itayose_time(now):
            if self.bid_order.is_live:
                self.bid_order = await self.client.cancel_order(
                    self.bid_order.order_id, timestamp=msg["timestamp"],
                    src_type="Rate", src_timestamp=msg["timestamp"], src_id=msg.get("universal_id", ""))
            if self.ask_order.is_live:
                self.ask_order = await self.client.cancel_order(
                    self.ask_order.order_id, timestamp=msg["timestamp"],
                    src_type="Rate", src_timestamp=msg["timestamp"], src_id=msg.get("universal_id", ""))
            return
        
        # OFI
        ofi = self._get_ofi(now)
        
        # 
        pos_skew = -self.cur_pos / self.max_position * self.position_decay * 10  # bps
        
        # OFIOFI===Ask
        ofi_skew = ofi * self.skew_factor * 10 if abs(ofi) > self.ofi_threshold else 0
        
        total_skew = ofi_skew + pos_skew
        
        bid_offset = self.base_offset - total_skew
        ask_offset = self.base_offset + total_skew
        
        new_bid = np.floor(self.mid * (1 - bid_offset / 1e4))
        new_ask = np.ceil(self.mid * (1 + ask_offset / 1e4))
        
        # 
        can_buy = self.cur_pos < self.max_position
        can_sell = self.cur_pos > -self.max_position
        
        # Bid
        if can_buy:
            need_update = not self.bid_order.is_live or abs(new_bid - self.bid_order.price) > 1
            if need_update:
                if self.bid_order.is_live:
                    self.bid_order = await self.client.cancel_order(
                        self.bid_order.order_id, timestamp=msg["timestamp"],
                        src_type="Rate", src_timestamp=msg["timestamp"], src_id=msg.get("universal_id", ""))
                if not self.bid_order.is_live:
                    self.bid_order = await self.client.send_order(
                        msg["timestamp"], self.sym, side=1, price=new_bid, amount=self.amount,
                        order_type=OrderType.Limit, src_type="Rate", src_timestamp=msg["timestamp"],
                        src_id=msg.get("universal_id", ""), misc="bid")
        elif self.bid_order.is_live:
            self.bid_order = await self.client.cancel_order(
                self.bid_order.order_id, timestamp=msg["timestamp"],
                src_type="Rate", src_timestamp=msg["timestamp"], src_id=msg.get("universal_id", ""))
        
        # Ask
        if can_sell:
            need_update = not self.ask_order.is_live or abs(new_ask - self.ask_order.price) > 1
            if need_update:
                if self.ask_order.is_live:

def prepare_backtest_data(st, et, venue="bitflyer", sym="FXBTCJPY"):
    """"""
    # Rate
    rate_fx = load_data(date=[st, et], table="Rate", venue=venue, symbol=sym)
    rate_spot = load_data(date=[st, et], table="Rate", venue=venue, symbol="BTCJPY")
    rate_binance = load_data(date=[st, et], table="Rate", venue="binance", symbol="BTCUSDT")
    rate_bitmex = load_data(date=[st, et], table="Rate", venue="bitmex", symbol="XBTUSD")
    usdjpy = load_data(date=[st, et], table="Rate", venue="gmo", symbol="USDJPY")
    
    # MarketTrade
    trades = load_data(date=[st, et], table="MarketTrade", venue=venue, symbol=sym)
    
    # MarketBook
    book = load_data(date=[st, et], table="MarketBook", venue=venue, symbol=sym)
    
    # UTC 10:00-10:20 = JST 19:00-19:20
    def filter_itayose(df, ts_col="timestamp"):
        if ts_col not in df.columns:
            ts_col = "market_created_timestamp"
        mask = ~((df[ts_col].dt.hour == 10) & (df[ts_col].dt.minute >= 0) & (df[ts_col].dt.minute <= 20))
        return df[mask].copy()
    
    rate_fx = filter_itayose(rate_fx)
    rate_spot = filter_itayose(rate_spot)
    trades = filter_itayose(trades, "market_created_timestamp")
    book = filter_itayose(book, "market_created_timestamp")
    
    # 
    rate_msgs = pd.concat([rate_fx, rate_spot, rate_binance, rate_bitmex, usdjpy]).assign(_data_type="Rate")
    rate_msgs = list(zip(rate_msgs.timestamp, rate_msgs.to_dict(orient="records")))
    
    # BTMarketTradeMarketBook
    bt_trades = list(zip(
        trades.market_created_timestamp,
        trades.assign(_data_type="BT_MarketTrade").to_dict(orient="records")
    ))
    
    bt_book = list(zip(
        book.market_created_timestamp,
        book.assign(_data_type="BT_MarketBook").to_dict(orient="records")
    ))
    
    # 
    messages = rate_msgs + bt_trades + bt_book
    messages = sorted(messages, key=lambda x: x[0])
    
    print(f"Total messages: {len(messages):,}")
    print(f"  - Rate: {len(rate_msgs):,}")
    print(f"  - MarketTrade: {len(bt_trades):,}")
    print(f"  - MarketBook: {len(bt_book):,}")
    
    return messages


async def run_backtest(strategy, messages, start_time=None):
    """"""
    client = strategy.client
    strategy.logger.setLevel(logging.CRITICAL)
    
    # MarketBook_latest_book
    for t, msg in messages:
        if msg.get("_data_type") == "BT_MarketBook":
            client._latest_book = msg
            break
    
    for i, (t, msg) in enumerate(tqdm(messages)):
        if start_time and t < start_time:
            continue
        await client.on_message(msg)
        await strategy.on_message(msg)
    
    return client


def analyze_results(client, fee_maker=0, fee_taker=0):
    """"""
    if len(client.transactions_frame) == 0:
        print("No trades executed.")
        return None
    
    # FIFO
    mat = calc_matching_table(client.transactions_frame)
    if len(mat) == 0:
        print("No matched trades.")
        return None
    
    # 
    mat["fee_bp"] = (
        mat.rinc_order_type.replace({"limit": fee_maker, "market": fee_taker}) +
        mat.rdec_order_type.replace({"limit": fee_maker, "market": fee_taker})
    )
    mat["pl_net"] = mat.pl_bp + mat.fee_bp
    mat["pl_net_jpy"] = mat.pl_net * mat.amount * mat.rinc_price / 1e4
    
    # 
    days = (mat.timestamp.max() - mat.timestamp.min()).days + 1
    
    stats = {
        "": len(mat),
        "1": len(mat) / days,
        "(bps)": mat.pl_net.sum(),
        "(bps)": mat.pl_net.mean(),
        "": (mat.pl_net > 0).mean(),
        "DD(bps)": mat.pl_net.cumsum().cummax().sub(mat.pl_net.cumsum()).max(),
        "Sharpe()": mat.groupby(mat.timestamp.dt.date).pl_net.sum().mean() / mat.groupby(mat.timestamp.dt.date).pl_net.sum().std() * np.sqrt(365) if mat.groupby(mat.timestamp.dt.date).pl_net.sum().std() > 0 else 0,
        "()": mat.horizon.mean(),
    }
    
    return mat, pd.Series(stats)
