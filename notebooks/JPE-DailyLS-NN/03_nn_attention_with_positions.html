def create_features_with_position(group):
    """"""
    g = group.sort_values('Date').copy()
    
    # 
    prev_close = g['Close'].shift(1)
    
    # === ===
    # 
    g['ret_1d'] = prev_close.pct_change(1)
    g['ret_5d'] = prev_close.pct_change(5)
    g['ret_10d'] = prev_close.pct_change(10)
    g['ret_20d'] = prev_close.pct_change(20)
    g['ret_60d'] = prev_close.pct_change(60)
    
    # 
    daily_ret = prev_close.pct_change(1)
    g['vol_5d'] = daily_ret.rolling(5).std().shift(1)
    g['vol_20d'] = daily_ret.rolling(20).std().shift(1)
    g['vol_60d'] = daily_ret.rolling(60).std().shift(1)
    
    # 
    g['vol_ratio_5_20'] = g['vol_5d'] / (g['vol_20d'] + 1e-8)
    
    # 
    prev_volume = g['Volume'].shift(1)
    g['vol_ma5'] = prev_volume.rolling(5).mean()
    g['vol_ma20'] = prev_volume.rolling(20).mean()
    g['vol_ratio'] = prev_volume / (g['vol_ma20'] + 1e-8)
    
    # 
    prev_high = g['High'].shift(1)
    prev_low = g['Low'].shift(1)
    g['high_20d'] = prev_high.rolling(20).max()
    g['low_20d'] = prev_low.rolling(20).min()
    g['price_position'] = (prev_close - g['low_20d']) / (g['high_20d'] - g['low_20d'] + 1e-8)
    
    # 
    g['ma5'] = prev_close.rolling(5).mean()
    g['ma20'] = prev_close.rolling(20).mean()
    g['ma60'] = prev_close.rolling(60).mean()
    g['ma_dev5'] = prev_close / g['ma5'] - 1
    g['ma_dev20'] = prev_close / g['ma20'] - 1
    g['ma_dev60'] = prev_close / g['ma60'] - 1
    
    # MA
    g['ma20_slope'] = g['ma20'].pct_change(5)
    g['ma60_slope'] = g['ma60'].pct_change(10)
    
    # RSI
    delta = prev_close.diff()
    gain = (delta.where(delta > 0, 0)).rolling(14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(14).mean()
    rs = gain / (loss + 1e-8)
    g['rsi_14'] = 100 - (100 / (1 + rs))
    
    # 
    g['prev_o2c'] = (g['Close'].shift(1) / g['Open'].shift(1) - 1)
    g['prev_overnight'] = (g['Open'].shift(1) / g['Close'].shift(2) - 1)
    g['prev_range'] = (prev_high - prev_low) / prev_close
    
    # ===  ===
    # 
    # margin_ratio, margin_balance 
    
    # 
    g['margin_ratio_chg'] = g['margin_ratio'].pct_change(5)  # 1
    g['short_vol_chg'] = g['ShrtVol'].pct_change(5)
    g['long_vol_chg'] = g['LongVol'].pct_change(5)
    
    # 
    g['short_ratio_chg'] = g['short_ratio'].pct_change(5)
    
    return g

class FeatureAttention(nn.Module):
    """Self-Attention"""
    def __init__(self, input_dim, num_heads=4, dropout=0.1):
        super().__init__()
        self.input_dim = input_dim
        self.num_heads = num_heads
        self.head_dim = max(1, input_dim // num_heads)
        
        # 
        self.embed_dim = self.head_dim * num_heads
        self.input_proj = nn.Linear(1, self.embed_dim)
        
        # Multi-head Self-Attention
        self.attention = nn.MultiheadAttention(
            embed_dim=self.embed_dim,
            num_heads=num_heads,
            dropout=dropout,
            batch_first=True
        )
        
        self.norm1 = nn.LayerNorm(self.embed_dim)
        self.norm2 = nn.LayerNorm(self.embed_dim)
        
        # Feed-forward
        self.ff = nn.Sequential(
            nn.Linear(self.embed_dim, self.embed_dim * 2),
            nn.GELU(),
            nn.Dropout(dropout),
            nn.Linear(self.embed_dim * 2, self.embed_dim)
        )
        
    def forward(self, x):
        # x: (batch, num_features)
        batch_size, num_features = x.shape
        
        #  (batch, num_features, embed_dim)
        x = x.unsqueeze(-1)  # (batch, num_features, 1)
        x = self.input_proj(x)  # (batch, num_features, embed_dim)
        
        # Self-Attention
        attn_out, attn_weights = self.attention(x, x, x)
        x = self.norm1(x + attn_out)
        
        # Feed-forward
        ff_out = self.ff(x)
        x = self.norm2(x + ff_out)
        
        # 
        x = x.mean(dim=1)  # (batch, embed_dim)
        
        return x, attn_weights


class AttentionRankPredictor(nn.Module):
    """Attention"""
    def __init__(self, input_dim, hidden_dims=[128, 64], num_heads=4, dropout=0.3):
        super().__init__()
        
        # Feature Attention
        self.feature_attention = FeatureAttention(input_dim, num_heads, dropout)
        embed_dim = self.feature_attention.embed_dim
        
        # MLP layers
        layers = []
        prev_dim = embed_dim
        
        for hidden_dim in hidden_dims:
            layers.extend([
                nn.Linear(prev_dim, hidden_dim),
                nn.BatchNorm1d(hidden_dim),
                nn.GELU(),
                nn.Dropout(dropout)
            ])
            prev_dim = hidden_dim
        
        layers.append(nn.Linear(prev_dim, 1))
        layers.append(nn.Sigmoid())
        
        self.mlp = nn.Sequential(*layers)
        
    def forward(self, x, return_attention=False):
        attn_out, attn_weights = self.feature_attention(x)
        out = self.mlp(attn_out)
        
        if return_attention:
            return out, attn_weights
        return out

def train_model(model, train_loader, val_loader, epochs=100, lr=0.001, patience=15):
    """"""
    optimizer = optim.AdamW(model.parameters(), lr=lr, weight_decay=1e-4)
    scheduler = optim.lr_scheduler.CosineAnnealingWarmRestarts(optimizer, T_0=20, T_mult=2)
    criterion = nn.MSELoss()
    
    best_val_loss = float('inf')
    best_model_state = None
    patience_counter = 0
    
    train_losses = []
    val_losses = []
    
    for epoch in range(epochs):
        # 
        model.train()
        train_loss = 0
        for X_batch, y_batch in train_loader:
            X_batch, y_batch = X_batch.to(device), y_batch.to(device)
            
            optimizer.zero_grad()
            pred = model(X_batch)
            loss = criterion(pred.squeeze(), y_batch)
            loss.backward()
            torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            optimizer.step()
            
            train_loss += loss.item()
        
        train_loss /= len(train_loader)
        train_losses.append(train_loss)
        scheduler.step()
        
        # 
        model.eval()
        val_loss = 0
        with torch.no_grad():
            for X_batch, y_batch in val_loader:
                X_batch, y_batch = X_batch.to(device), y_batch.to(device)
                pred = model(X_batch)
                loss = criterion(pred.squeeze(), y_batch)
                val_loss += loss.item()
        
        val_loss /= len(val_loader)
        val_losses.append(val_loss)
        
        if val_loss < best_val_loss:
            best_val_loss = val_loss
            best_model_state = model.state_dict().copy()
            patience_counter = 0
        else:
            patience_counter += 1
        
        if (epoch + 1) % 10 == 0:
            print(f'Epoch {epoch+1}: Train={train_loss:.6f}, Val={val_loss:.6f}, LR={scheduler.get_last_lr()[0]:.6f}')
        
        if patience_counter >= patience:
            print(f'Early stopping at epoch {epoch+1}')
            break
    
    model.load_state_dict(best_model_state)
    return model, train_losses, val_losses

def calc_long_short_returns(df_day, n_stocks=30):
    df_sorted = df_day.sort_values('pred_rank', ascending=False)
    long_stocks = df_sorted.head(n_stocks)
    short_stocks = df_sorted.tail(n_stocks)
    
    long_ret = long_stocks['O2C'].mean()
    short_ret = short_stocks['O2C'].mean()
    
    return {
        'long_ret': long_ret,
        'short_ret': short_ret,
        'ls_ret': long_ret - short_ret
    }

def calc_stats(returns, name):
    n = len(returns)
    total = (1 + returns).prod() - 1
    annual = (1 + total) ** (252 / n) - 1
    vol = returns.std() * np.sqrt(252)
    sharpe = annual / vol if vol > 0 else 0
    cum = (1 + returns).cumprod()
    dd = (cum / cum.cummax() - 1).min()
    win = (returns > 0).mean()
    
    return {
        'Name': name,
        'Total (%)': total * 100,
        'Annual (%)': annual * 100,
        'Vol (%)': vol * 100,
        'Sharpe': sharpe,
        'MaxDD (%)': dd * 100,
        'WinRate (%)': win * 100,
        'Days': n
    }

stats = [
    calc_stats(df_returns['ls_ret'], 'Long-Short'),

    calc_stats(df_returns['short_ret'], 'Short Only'),

]
