def create_features_no_leak(group):
    """
    
    
    TT-1
    """
    g = group.sort_values('Date').copy()
    
    # 
    prev_close = g['Close'].shift(1)
    
    # - 
    # T = T-1
    g['ret_1d'] = prev_close.pct_change(1)  # T-2T-1
    g['ret_5d'] = prev_close.pct_change(5)  # T-6T-1
    g['ret_10d'] = prev_close.pct_change(10)
    g['ret_20d'] = prev_close.pct_change(20)
    
    #  - 
    daily_ret = prev_close.pct_change(1)
    g['vol_5d'] = daily_ret.rolling(5).std().shift(1)  # T-6~T-2
    g['vol_10d'] = daily_ret.rolling(10).std().shift(1)
    g['vol_20d'] = daily_ret.rolling(20).std().shift(1)
    
    #  - 
    prev_volume = g['Volume'].shift(1)
    g['vol_ma5'] = prev_volume.rolling(5).mean()
    g['vol_ma20'] = prev_volume.rolling(20).mean()
    g['vol_ratio'] = prev_volume / g['vol_ma20']
    
    # 
    prev_turnover = g['Turnover'].shift(1)
    g['turnover_ma5'] = prev_turnover.rolling(5).mean()
    g['turnover_ma20'] = prev_turnover.rolling(20).mean()
    
    #  - 
    prev_high = g['High'].shift(1)
    prev_low = g['Low'].shift(1)
    g['high_20d'] = prev_high.rolling(20).max()
    g['low_20d'] = prev_low.rolling(20).min()
    g['price_position'] = (prev_close - g['low_20d']) / (g['high_20d'] - g['low_20d'] + 1e-8)
    
    #  - 
    g['ma5'] = prev_close.rolling(5).mean()
    g['ma10'] = prev_close.rolling(10).mean()
    g['ma20'] = prev_close.rolling(20).mean()
    g['ma60'] = prev_close.rolling(60).mean()
    g['ma_dev5'] = prev_close / g['ma5'] - 1
    g['ma_dev10'] = prev_close / g['ma10'] - 1
    g['ma_dev20'] = prev_close / g['ma20'] - 1
    g['ma_dev60'] = prev_close / g['ma60'] - 1
    
    # RSI - 
    delta = prev_close.diff()
    gain = (delta.where(delta > 0, 0)).rolling(14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(14).mean()
    rs = gain / (loss + 1e-8)
    g['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Open to Close 
    g['prev_o2c'] = (g['Close'].shift(1) / g['Open'].shift(1) - 1)
    
    # 
    g['prev_overnight'] = (g['Open'].shift(1) / g['Close'].shift(2) - 1)
    
    # 
    g['prev_range'] = (prev_high - prev_low) / prev_close
    
    return g

class RankPredictionMLP(nn.Module):
    """MLP"""
    def __init__(self, input_dim, hidden_dims=[128, 64, 32], dropout=0.3):
        super().__init__()
        
        layers = []
        prev_dim = input_dim
        
        for hidden_dim in hidden_dims:
            layers.extend([
                nn.Linear(prev_dim, hidden_dim),
                nn.BatchNorm1d(hidden_dim),
                nn.ReLU(),
                nn.Dropout(dropout)
            ])
            prev_dim = hidden_dim
        
        layers.append(nn.Linear(prev_dim, 1))
        layers.append(nn.Sigmoid())  # 0-1
        
        self.network = nn.Sequential(*layers)
    
    def forward(self, x):
        return self.network(x)

def train_model(model, train_loader, val_loader, epochs=50, lr=0.001, patience=10):
    """"""
    optimizer = optim.Adam(model.parameters(), lr=lr, weight_decay=1e-5)
    scheduler = optim.lr_scheduler.ReduceLROnPlateau(optimizer, mode='min', factor=0.5, patience=5)
    criterion = nn.MSELoss()
    
    best_val_loss = float('inf')
    best_model_state = None
    patience_counter = 0
    
    train_losses = []
    val_losses = []
    
    for epoch in range(epochs):
        # 
        model.train()
        train_loss = 0
        for X_batch, y_batch in train_loader:
            X_batch, y_batch = X_batch.to(device), y_batch.to(device)
            
            optimizer.zero_grad()
            pred = model(X_batch)
            loss = criterion(pred.squeeze(), y_batch)
            loss.backward()
            optimizer.step()
            
            train_loss += loss.item()
        
        train_loss /= len(train_loader)
        train_losses.append(train_loss)
        
        # 
        model.eval()
        val_loss = 0
        with torch.no_grad():
            for X_batch, y_batch in val_loader:
                X_batch, y_batch = X_batch.to(device), y_batch.to(device)
                pred = model(X_batch)
                loss = criterion(pred.squeeze(), y_batch)
                val_loss += loss.item()
        
        val_loss /= len(val_loader)
        val_losses.append(val_loss)
        
        scheduler.step(val_loss)
        
        # Early stopping
        if val_loss < best_val_loss:
            best_val_loss = val_loss
            best_model_state = model.state_dict().copy()
            patience_counter = 0
        else:
            patience_counter += 1
        
        if (epoch + 1) % 10 == 0:
            print(f'Epoch {epoch+1}: Train Loss={train_loss:.6f}, Val Loss={val_loss:.6f}')
        
        if patience_counter >= patience:
            print(f'Early stopping at epoch {epoch+1}')
            break
    
    model.load_state_dict(best_model_state)
    return model, train_losses, val_losses

def calc_performance_stats(returns, name='Strategy'):
    n_days = len(returns)
    annual_factor = 252
    
    total_ret = (1 + returns).prod() - 1
    annual_ret = (1 + total_ret) ** (annual_factor / n_days) - 1
    annual_vol = returns.std() * np.sqrt(annual_factor)
    sharpe = annual_ret / annual_vol if annual_vol > 0 else 0
    
    cum_ret = (1 + returns).cumprod()
    rolling_max = cum_ret.cummax()
    drawdown = (cum_ret - rolling_max) / rolling_max
    max_dd = drawdown.min()
    
    win_rate = (returns > 0).mean()
    
    return {
        'Name': name,
        'Total Return (%)': total_ret * 100,
        'Annual Return (%)': annual_ret * 100,
        'Annual Vol (%)': annual_vol * 100,
        'Sharpe Ratio': sharpe,
        'Max Drawdown (%)': max_dd * 100,
        'Win Rate (%)': win_rate * 100,
        'Trading Days': n_days
    }
